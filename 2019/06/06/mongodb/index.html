node mongoose 使用


©敲代码的小新

1.连接数据库

const mongoose = require('mongoose');

// 数据库连接 27017是mongodb数据库的默认端口
mongoose.connect('mongodb://localhost/playground', { useNewUrlParser: true, useUnifiedTopology: true })
.then(() => console.log('数据库连接成功'))
.catch(() => console.log('数据库连接失败'));

2.创建集合规则 mongoose.Schema

const courseSchema = new mongoose.Schema({
// name: String,
name: {
    type: String, //字段类型
    required: true, //必填字段
    minlength: 3, //最小字符长度
    maxlength: 10, //最大字符长度
    trim: true //去掉空格
},
author: {
    type: String,
    required: [true, '作者字段不能为空'], //后面可以添加字段错误信息
    minlength: [3, '最小字符长度为3'],
    maxlength: [10, '最大字符长度为10'],
    trim: true
},
age: {
    type: Number,
    min: 0,
    max: 120
},
publishDate: {
    type: Date,
    default: Date.now //默认值
},
category: {
    type: String,
    // 枚举 列举当前字段可以拥有的值， 前台传入的时候 必须传入规定的数值才能插入成功
    // enum: ['java', 'android', 'nodejs', 'ios']
    enum: {
        values: ['java', 'android', 'nodejs', 'ios'],
        message: '分类名称要在规定的范围之内'
    }
},
sex: {
    type: Number,
    // 自定义验证规则
    validate: {
        validator: v => {
            return v && (v == 0 || v == 1);
        },
        message: '性别请传入0 或 1';
    }
},
isPublished: Boolean

});

3.创建集合 返回集合构造函数

//参数 第一个参数为表名称 规则 首字母大写，并且数据库创建的表默认会加s ；第二个参数为集合规则
const User = mongoose.model('User', userSchema);

4.插入数据
save插入一条数据
create插入一条数据

a.   使用结合构造函数创建一条数据
     const course = new Course({
         name: 'android 基础',
         author: 'xiaoxin',
         isPublished: true
     });
     插入一条数据
     course.save();
b.数据库操作支持Promise
 Course.create({
         name: 'javaScript123 基础',
         author: 'xiaoxin',
         isPublished: true
     })
     .then(doc => console.log(doc))
     .catch(err => console.log(err));

5.查询数据

//查询所有数据
Course.find()
      .then(doc => console.log(doc))
      .catch(err => console.log(err));

//安装规则查询所有数据
// $gt 大于  $lt 小于
// $in 包含
Course.findOne({ _id: '5df5876fbbf8f2108861c6ea' })
      .then(result => console.log(result))
      .catch(err => console.log(err));

 //select 里面可以选择要查询的字段  -代表不查询的字段
 Course.find().select('name author -_id').then(result => console.log(result));
 //按照年龄进行升序排序 传入-age代表降序排序
 Course.find().sort('age').then(result => console.log(result));

// skip 跳过第一条数据  limit 返回2条结果  分页使用
 Course.find().skip(1).limit(2).then(result => console.log(result));


findOne查询一条数据
// 查询和id想匹配的第一条数据  结果是对象类型
 Course.findOne({ _id: '5df5876fbbf8f2108861c6ea' })
     .then(result => console.log(result))
     .catch(err => console.log(err));

6.更新数据

// 更新数据
// updateOne({查询条件},{更改数据})
// 更新一条数据
Course.updateOne({ name: 'android 基础' }, { name: 'ios 基础' })
      .then(result => console.log(result));

// 更新多条数据
Course.updateMany({ author: 'xiaoxin' }, { author: '小新' })
      .then(result => console.log(result));

7.删除数据

// 根据条件删除一条数据 如果条件查询多条数据  只删除第一条数据
Course.findOneAndDelete({ _id: '5df589bcdc8ad14b2c9ea458' })
     .then(result => console.log(result))
     .catch(err => console.log(err));

//根据条件全部删除
 Course.deleteMany({}).then(result => console.log(result));

8.关联查询

字段需要 filed: {
            type: mongoose.Schema.Types.ObjectId,
            ref: 'User'
        }
查询使用populate

// 用户集合规则
const userSchema = new mongoose.Schema({
    name: {
        type: String,
        required: true
    }
});
// 文章集合规则
const postSchema = new mongoose.Schema({
    title: {
        type: String
    },
    author: {
        type: mongoose.Schema.Types.ObjectId,
        ref: 'User'
    }
});
// 用户集合
const User = mongoose.model('User', userSchema);
// 文章集合
const Post = mongoose.model('Post', postSchema);

// 创建用户
// User.create({name: 'xiaoxin'}).then(result => console.log(result));
// 创建文章
// Post.create({titile: '123', author: '000caae2c4e4081c28439791'}).then(result => console.log(result));
Post.find().populate('author').then(result => console.log(result))

9.获取数据库操作错误对象

Post.create({title:'aa', age: 60, category: 'java', author: 'bd'})
    .then(result => console.log(result))
    .catch(error => {
        // 获取错误信息对象
        const err = error.errors;
        // 循环错误信息对象
        for (var attr in err) {
            // 将错误信息打印到控制台中
            console.log(err[attr]['message']);
        }
    })
