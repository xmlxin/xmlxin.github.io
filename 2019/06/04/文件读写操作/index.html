nodejs fs文件读写使用


©敲代码的小新
##原生js写法
1.引入fs模块

const fs = require('fs');

2.读取操作

 fs.readFile('./02.js', 'utf-8', (err, result) => {
    if (err != null) {
       console.log('err不为空代表读取失败');
    } else {
        console.log('读取成功'+result);
    }
});

3.写入操作

//文件不存在会自动创建
fs.writeFile('./year.txt', '写入操作', (err, result) => {
    if (err != null) {
        console.log('err不为空代表写入失败');
    } else {
        console.log('写入成功');
    }
});

##封装成promisify对象写法
##支持异步方法获取返回值

方法前面添加async关键字 读写操作前面添加await关键字

//1.封装fs为promisify对象
const promisify = require('util').promisify;
const readfile = promisify(fs.readFile);

app.use('/index', async(req, res, next) => {
   try {
        let result = await readfile('./02.js');
        res.send(result);
     } catch (err) {
         next(err);
     }
});
